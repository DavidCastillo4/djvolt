CREATE TABLE reservation (
reservationPk INTEGER GENERATED ALWAYS AS IDENTITY (START WITH 1001 INCREMENT BY 1) PRIMARY KEY,
date             DATE NULL,
slot             VARCHAR(2) NULL CHECK (Slot IN ('am', 'pm')),
firstName varchar(100) null,
lastName varchar(100) null,
mobile           VARCHAR(20) NULL,
email            VARCHAR(255) NULL,
Address1 varchar(500) NULL,
notes varchar(5000) null,
entryCode varchar(25) null,
entryCodeSentDateTime TIMESTAMP null,
homeOwner BOOLEAN null,
status varchar(50) NULL,
residentMessage varchar(5000) null,
adminNotes varchar(5000),
hashId varchar(200),
lastModifiedUser varchar(100) null,
lastModifiedDateTime TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
);
